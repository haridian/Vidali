1 var mapView = Backbone.View.extend({
  2   el: $('#container'),
  3   map: null,
  4   model: new mapModel(),
  5   initialize: function () {    
  6   },
  7   render: function () {
  8     var latitude = parseFloat(localStorage.getItem('latitude'));
  9     var longitude = parseFloat(localStorage.getItem('longitude'));
 10     if(this.model.get("map") == "default"){      
 11         this.map = new ol.Map({
 12             target: 'map',
 13             layers: [
 14               new ol.layer.Tile({
 15                 source: new ol.source.OSM()
 16               })
 17             ],
 18             renderers: ol.RendererHints.createFromQueryData(),
 19             view: new ol.View2D({
 20                 center: ol.proj.transform([longitude,latitude], 'EPSG:4326', 'EPSG:3857'),
 21                 zoom: 16
 22             })
 23         });
 24         var geolocation = new ol.Geolocation();
 25         geolocation.bindTo('projection', this.map.getView());
 26         geolocation.setTracking(true);
 27         var marker = new ol.Overlay({
 28           element: document.getElementById('position'),
 29           position: ol.proj.transform([longitude,latitude], 'EPSG:4326', 'EPSG:3857'),
 30           positioning: ol.OverlayPositioning.CENTER_CENTER
 31         });
 32         this.map.addOverlay(marker);
 33         // bind the marker position to the device location.
 34         marker.bindTo('position', geolocation);
 35         return true;
 36     }
 37   }
 38 });
